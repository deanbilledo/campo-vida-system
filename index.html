${product.name}

₱${product.price.toFixed(2)} per kg

${product.description.substring(0, 50)}...

`; grid.appendChild(productCard); }); } function filterProducts(category) { if (category === 'all') { renderProducts(); } else { const filtered = products.filter(p => p.category === category); renderProducts(filtered); } } function openProductModal(product) { selectedProduct = product; document.getElementById('modalTitle').textContent = product.name; document.getElementById('modalImage').src = product.image; document.getElementById('modalImage').alt = product.name; document.getElementById('modalPrice').textContent = `₱${product.price.toFixed(2)}`; document.getElementById('modalDescription').textContent = product.description; document.getElementById('quantity').textContent = '1'; updateTotalPrice(); updateBuyerGuidance(); document.getElementById('productModal').classList.add('show'); } function closeProductModal() { document.getElementById('productModal').classList.remove('show'); selectedProduct = null; } function updateTotalPrice() { if (!selectedProduct) return; const qty = parseInt(document.getElementById('quantity').textContent); const total = selectedProduct.price * qty; document.getElementById('totalPrice').textContent = total.toFixed(2); } function updateBuyerGuidance() { const guidance = document.getElementById('buyerGuidance'); const orderProgress = document.getElementById('orderProgress'); if (!currentUser) { guidance.classList.add('hidden'); return; } if (userOrderCount < 5) { guidance.classList.remove('hidden'); orderProgress.textContent = `Orders completed: ${userOrderCount}/5`; if (userOrderCount === 0) { guidance.querySelector('p').textContent = 'Welcome to Campo Vida! For your first purchase, you\'ll need to pay online. After 5 successful orders, you\'ll unlock Cash on Delivery (COD) option.'; } else { guidance.querySelector('p').textContent = `You've completed ${userOrderCount}/5 orders. ${5 - userOrderCount} more orders to unlock COD option.`; } } else { guidance.classList.add('hidden'); } } function buyNow() { if (!currentUser) { openLoginModal(); return; } if (!selectedProduct) return; const qty = parseInt(document.getElementById('quantity').textContent); // Show guidance for first-time buyers if (userOrderCount === 0) { if (confirm('Welcome to Campo Vida! For your first purchase, you\'ll need to pay online. After 5 successful orders, you\'ll unlock Cash on Delivery (COD) option. Continue to purchase?')) { processPurchase(qty); } } else if (userOrderCount < 5) { if (confirm(`You've completed ${userOrderCount}/5 orders. ${5 - userOrderCount} more orders to unlock COD option. Continue to purchase?`)) { processPurchase(qty); } } else { processPurchase(qty); } } function processPurchase(qty) { // Simulate purchase userOrderCount++; closeProductModal(); showNotification(`Purchase successful! You bought ${qty} kg of ${selectedProduct.name}`); updateBuyerGuidance(); } function addToCart() { if (!currentUser) { openLoginModal(); return; } if (!selectedProduct) return; const qty = parseInt(document.getElementById('quantity').textContent); const existingItem = cart.find(item => item.id === selectedProduct.id); if (existingItem) { existingItem.quantity += qty; } else { cart.push({ ...selectedProduct, quantity: qty }); } updateCartCount(); closeProductModal(); showNotification(`Added ${qty} kg of ${selectedProduct.name} to cart`); } function updateCartCount() { const count = cart.reduce((sum, item) => sum + item.quantity, 0); const countEl = document.getElementById('cartCount'); countEl.textContent = count; countEl.classList.toggle('show', count > 0); } function openCartModal() { renderCartItems(); document.getElementById('cartModal').classList.add('show'); } function closeCartModal() { document.getElementById('cartModal').classList.remove('show'); } function renderCartItems() { const container = document.getElementById('cartItems'); const total = cart.reduce((sum, item) => sum + (item.price * item.quantity), 0); if (cart.length === 0) { container.innerHTML = '

Your cart is empty

'; document.getElementById('cartTotal').textContent = '₱0.00'; return; } container.innerHTML = cart.map(item => `

${item.name}

₱${item.price.toFixed(2)} × ${item.quantity} kg

₱${(item.price * item.quantity).toFixed(2)}

`).join(''); document.getElementById('cartTotal').textContent = `₱${total.toFixed(2)}`; } function removeFromCart(productId) { cart = cart.filter(item => item.id !== productId); updateCartCount(); renderCartItems(); showNotification('Item removed from cart'); } function openLoginModal() { document.getElementById('loginModal').classList.add('show'); } function closeLoginModal() { document.getElementById('loginModal').classList.remove('show'); } function showNotification(message) { const notification = document.getElementById('notification'); document.getElementById('notificationText').textContent = message; notification.classList.add('show'); setTimeout(() => { notification.classList.remove('show'); }, 3000); } // Checkout functionality document.getElementById('checkoutBtn').addEventListener('click', function() { if (cart.length === 0) { showNotification('Your cart is empty'); return; } const total = cart.reduce((sum, item) => sum + (item.price * item.quantity), 0); if (userOrderCount < 5) { if (confirm(`Total: ₱${total.toFixed(2)}\n\nNote: Online payment required for ${userOrderCount < 1 ?
